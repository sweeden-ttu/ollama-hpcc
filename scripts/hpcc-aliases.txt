# -----------------------------------------------------------------------------
# Connection
# -----------------------------------------------------------------------------
alias hpcc='ssh -q -i /Users/owner/.ssh/id_rsa sweeden@login.hpcc.ttu.edu'
alias hpcc-login='ssh -q -i /Users/owner/.ssh/id_rsa sweeden@login.hpcc.ttu.edu'

# -----------------------------------------------------------------------------
# Environment introspection (node, ollama_host, ollama_base_url, model, port)
# -----------------------------------------------------------------------------
hpcc-info() {
  ssh -q sweeden@login.hpcc.ttu.edu "squeue -u \$USER"
  echo ""
  echo "=== Latest Ollama job info (if any) ==="
  ssh -q sweeden@login.hpcc.ttu.edu 'ls -la ~/ollama-logs/*.info 2>/dev/null || echo "No job info files found"' 
}

hpcc-latest-log() {
  local job_status
  job_status=$(ssh -q sweeden@login.hpcc.ttu.edu "squeue -u \$USER -o '%T' -h | head -1")
  if [[ "$job_status" != "RUNNING" ]]; then
    echo "Job status: $job_status (not RUNNING)"
    echo "Run 'hpcc-info' to check queue status"
    return 1
  fi
  ssh -q sweeden@login.hpcc.ttu.edu 'latest=$(ls -t ~/ollama-logs/*.info | head -1) && tail -20 "$latest"'
}

# -----------------------------------------------------------------------------
# Job queue and control
# -----------------------------------------------------------------------------
hpcc-status() {
  ssh -q sweeden@login.hpcc.ttu.edu "squeue -u \$USER"
}
alias hpcc-jobs='hpcc-status'

hpcc-kill() {
  if [[ -z "$1" ]]; then
    echo "Usage: hpcc-kill JOBID"
    return 1
  fi
  ssh -q sweeden@login.hpcc.ttu.edu "scancel $1"
}

# -----------------------------------------------------------------------------
# Tunnels — do not use the default Ollama port; use these to expose server port locally
# Usage: hpcc-tunnel PORT NODE   e.g. hpcc-tunnel 34935 cpu-01-42
# -----------------------------------------------------------------------------
hpcc-tunnel() {
  if [[ $# -lt 2 ]]; then
    echo "Usage: hpcc-tunnel PORT NODE"
    echo "Example: hpcc-tunnel 34935 cpu-01-42"
    return 1
  fi
  local port=$1 node=$2
  ssh -q sweeden@login.hpcc.ttu.edu -L "${port}:${node}:${port}" -N
}

hpcc-tunnel-jump() {
  if [[ $# -lt 2 ]]; then
    echo "Usage: hpcc-tunnel-jump PORT NODE"
    echo "Example: hpcc-tunnel-jump 34935 cpu-01-42 (local → login → compute)"
    return 1
  fi
  local port=$1 node=$2
  ssh -q sweeden@login.hpcc.ttu.edu -L "${port}:${node}:${port}" -N
}

ollama-tunnel() {
  if [[ $# -lt 3 ]]; then
    echo "Usage: ollama-tunnel NODE PORT MODEL"
    echo "Example: ollama-tunnel gpu-20-1 33195 granite4:3b"
    echo "         ollama-tunnel cpu-01-42 34935 deepseek-coder:6.7b"
    return 1
  fi
  local node=$1 port=$2 model=$3
  echo "=== Creating tunnel ==="
  echo "Node: $node"
  echo "Port: $port" 
  echo "Model: $model"
  echo "Command: ssh -L ${port}:${node}:${port} hpcc-login -o ServerAliveInterval=60 -o ServerAliveCountMax=3 -N -f $node"
  echo "======================="
  ssh -L "${port}:${node}:${port}" hpcc-login -o ServerAliveInterval=60 -o ServerAliveCountMax=3 -N -f "$node"
  echo "Tunnel started. Test with: curl http://localhost:${port}/api/tags"
}
